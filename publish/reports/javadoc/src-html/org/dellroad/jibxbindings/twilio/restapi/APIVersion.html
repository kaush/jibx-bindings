008package org.dellroad.jibxbindings.twilio.restapi;
009
010import org.dellroad.jibxbindings.ParseUtil;
011import org.dellroad.jibxbindings.XMLEnum;
012import org.jibx.runtime.JiBXParseException;
013
014/**
015 * API versions. Enum values are ordered from oldest to most recent.
016 */
017public enum APIVersion implements XMLEnum {
018    V_2008_08_01("2008-08-01"),
019    V_2010_04_01("2010-04-01");
020
021    /**
022     * The API version that this library version adheres to (always the first enum value).
023     */
024    public static final APIVersion CURRENT = APIVersion.values()[0];
025
026    private final String xmlName;
027
028    private APIVersion(String xmlName) {
029        this.xmlName = xmlName;
030    }
031
032    @Override
033    public String getXMLName() {
034        return this.xmlName;
035    }
036
037    public static APIVersion deserializeXML(String string) throws JiBXParseException {
038        return ParseUtil.deserializeXMLEnum(string, APIVersion.values());
039    }
040}
